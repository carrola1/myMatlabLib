1function [newerrs] = myBERToolMex(msg,siglen,xFd,xInterp,dly,finalBit,HdTx,HdRx,HdInterp,snr)
 2
 3coder.extrinsic('awgn','intdump','rectpulse'); 
 4
 5msgUpSamp   = rectpulse(msg,xFd);
 6
 7msgFilt    = filter(HdTx,1,msgUpSamp);         % TX FIR Filter
 8
 9chlsig     = awgn(msgFilt, snr, 'measured');   % Add noise
10
11rxFilt     = filter(HdRx,1,chlsig);            % RX FIR Filter
12
13% Interpolate and adjust for delay
14rxInterp       = filter(HdInterp,1,rxFilt);
15rxInterpNew    = zeros(1,siglen*xFd*xInterp - dly - finalBit);
16rxInterpNew    = rxInterp(dly+1:siglen*xFd*xInterp-finalBit);
17
18% Bit Sync
19decodmsg = intdump(rxInterpNew,xFd*xInterp);      % Integrate and dump
20decodmsg = sign(decodmsg);                     % Demodulate
21
22newerrs = sum(abs(msg(1:length(msg)-(length(msg)-length(decodmsg))) - decodmsg))/2; % Errors in this trial
